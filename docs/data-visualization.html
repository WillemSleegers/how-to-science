3 Data Visualization

3.1 Bar charts 

p <- ggplot(mpg, aes(x = class, color = class, fill = class)) +
  geom_bar(alpha = .75) +
  scale_color_viridis_d() +
  scale_fill_viridis_d() +
  theme_minimal()

ggplotly(p)

3.2 Line charts 

p <- ggplot(economics, aes(x = date, y = unemploy)) + 
  geom_line(alpha = .75) +
  theme_minimal()

ggplotly(p)
p <- ggplot(economics_long, aes(x = date, y = value01, 
    color = variable)) +
  geom_line(alpha = .75) +
  scale_color_viridis_d() +
  theme_minimal()

ggplotly(p)
